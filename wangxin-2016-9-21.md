package text2;

public class Dog extends Pet{
	
	String strain;
	
	Dog(){
		
	}
	
	Dog(String name,String strain){
		this.name = name;
		this.health = 100;
		this.intimacy = 0;
		this.strain = strain;
	}
	
	public void catchingFlyDisc(){
		health = health - 10;
		intimacy = intimacy + 5;
	}
	public void print(){
		System.out.print("我的名字是"+name+",我的健康值是"+health+",我与主人的亲密度是 "+intimacy);
	}

}
